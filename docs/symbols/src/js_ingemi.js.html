1 (function (){
  2 /**
  3  * Ingemi(nate) - An iterative Mandelbrot set generator in JavaScript.
  4  * @name Ingemi
  5  * @class Ingemi
  6  * @constructor
  7  * @param {Object} args An argument object to override most defaults. Valid properties are listed below.
  8  * @property {Integer} upscale Used to strech internal canvas dimensions for fast, low-resolution renders.
  9  * @property {Float} rangeLeft Width of coordinate system mapped to the canvas (smaller numbers produce zoom).
 10  * @property {Float} rangeTop Height of coordinate system mapped to the canvas.
 11  * @property {Float} offsetLeft Horizontal offset of the coordinate system relative to the center of the canvas. Units are the same as rangeLeft.
 12  * @property {Float} offsetTop Vertical offset of the coordinate system relative to the center of the canvas.
 13  * @property {Integer} maxIteration Maximum iteration used in escape-velocity calculations. High numbers produce greater color differentiation.
 14  * @property {Integer} blockSize Number of pixels to render concurrently. Higher values may increase performance at the cost of browser stability.
 15  */
 16 function Ingemi (args) {
 17 
 18     /** Avoid errors if no arguments are passed */
 19     args = args || {};
 20 
 21     /** Bootstrap the page */
 22     this.ensureParent();
 23     this.makeCanvas();
 24     
 25     /** Set viewport and rendering defaults */
 26     this.setDefaults(args);
 27 
 28     /** Set internal canvas scale */
 29     this.scaleCanvas();
 30 
 31     /** Initialize zoom controller - TODO generalize for different fractals */
 32     this.zoomer = new IngemiZoom(this);
 33 
 34     /** Render the current view */
 35     this.render();
 36 }
 37 
 38 /**
 39  * Get and ensure the container for ingemi. Size and position of the canvas element will be
 40  *     determined from this element.
 41  * @throws Error if markup doesn't contain a node with id 'ingemi'
 42  */
 43 Ingemi.prototype.ensureParent = function () {
 44     this.parentDiv = document.getElementById('ingemi');
 45     if (!this.parentDiv) {
 46         throw new Error('You must include "<div id=\'ingemi\'></div>" in your markup');
 47     }
 48 };
 49 
 50 /**
 51  * Create a canvas in the parent div and inherit its size
 52  */
 53 Ingemi.prototype.makeCanvas = function () {
 54     this.canvas = document.createElement('canvas');
 55     this.parentDiv.appendChild(this.canvas);
 56     this.context = this.canvas.getContext('2d');
 57     this.clientWidth = this.parentDiv.clientWidth;
 58     this.clientHeight = this.parentDiv.clientHeight;
 59     this.canvas.style.width = this.clientWidth + 'px';
 60     this.canvas.style.height =  this.clientHeight + 'px';
 61 };
 62 
 63 /**
 64  * @param {Object} args
 65  */
 66 Ingemi.prototype.setDefaults = function (args) {
 67     this.upscale = args['upscale'] || 1;
 68     this.rangeLeft = args['rangeLeft'] || 1;
 69     this.rangeTop = args['rangeTop'] || 1;
 70     this.offsetLeft = args['offsetLeft'] || 0;
 71     this.offsetTop = args['offsetTop'] || 0;
 72     this.maxIteration = args['maxIteration'] || 255;
 73     this.blockSize = args['blockSize'] || 2500;
 74     this.onrender = (typeof args['onrender'] === 'function') ? args['onrender'] : null;
 75 };
 76 
 77 /**
 78  * Set the internal size of the canvas element
 79  */
 80 Ingemi.prototype.scaleCanvas = function () {
 81     this.width = this.canvas.width = Math.floor(this.clientWidth / this.upscale);
 82     this.height = this.canvas.height = Math.floor(this.clientHeight / this.upscale);
 83     this.image = this.context.createImageData(this.width, this.height);
 84     this.totalPixels = this.width * this.height;
 85 };
 86 
 87 /**
 88  * Render the entire scene using the current viewport and context
 89  */
 90 Ingemi.prototype.render = function () {
 91     this.timer = new Date().getTime();
 92     this.renderedPixels = 0;
 93     this.renderedPixelsInBlock = 0;
 94     this.blockOffset = 0;
 95     this.renderBlock();
 96 };
 97 
 98 /**
 99  * Render one row asynchronously - TODO convert to static sized blocks
100  */
101 Ingemi.prototype.renderBlock = function () {
102     var i;
103     var lim = (this.blockSize + this.blockOffset > this.totalPixels) ? this.totalPixels - this.blockOffset : this.blockSize;
104     for (i = 0; i < lim; i += 1) {
105         var pos = this.blockOffset + i;
106         this.setPixel(pos % this.width, Math.floor(pos/this.width));
107     }
108 };
109 
110 /**
111  * Asynchronously determine and set the value for the specified pixel
112  *     and internally check if we are done rendering either a block or image
113  * @param {Integer} left
114  * @param {Integer} top
115  */
116 Ingemi.prototype.setPixel = function (left, top) {
117     var _this = this;
118     setTimeout(function () {
119         var pos = _this.gridToLine(left, top) * 4;
120         var value = _this.getValue(left, top);
121         _this.setPixelColor(pos, value);
122         _this.updateCounters();
123     }, 0);
124 };
125 
126 /**
127  * Convert cartesian coordinates to a one-dimensional index
128  * @param {Integer} left
129  * @param {Integer} top
130  */
131 Ingemi.prototype.gridToLine = function (left, top) {
132     return top * this.width + left;
133 };
134 
135 /**
136  * Main logic for Mandelbrot generation
137  *     - TODO Abstract out for more fractal types
138  * @param {Integer} left
139  * @param {Integer} top
140  */
141 Ingemi.prototype.getValue = function (left, top) {
142     var scaledX = (this.rangeLeft * 3.5 * left / this.width) - 2.5 + this.offsetLeft;
143     var scaledY = (this.rangeTop * 2 * top / this.height) - 1 + this.offsetTop;
144     if (this.isInCartoid(scaledX, scaledY)) {   // Optimize against inner cartoid
145         return this.maxIteration;
146     } else {
147         var x = 0;
148         var y = 0;
149         var iteration = 0;
150         var xtemp;
151         while (x*x + y*y < 4 && iteration < this.maxIteration) {
152             xtemp = x*x - y*y + scaledX;
153             y = 2*x*y + scaledY;
154             x = xtemp;
155             iteration += 1;
156         }
157         return iteration;
158     }
159 };
160 
161 /**
162  * Simple optimization to prevent computing to maximum iteration in the center
163  *     of the unzoomed Mandelbrot set
164  * @param {Float} left
165  * @param {Float} top
166  */
167 Ingemi.prototype.isInCartoid = function (left, top) {
168     var p = Math.pow(left - 0.25, 2) + (top * top);
169     var q = Math.sqrt(p);
170     return left <= q - (2 * p) + 0.25;
171 };
172 
173 /**
174  * Map an integer [0...maxIteration] into some rgb spectrum
175  *     and write it to the imageData array
176  * @param {Integer} pos The linear position of the pixel being modified
177  * @param {Integer} value The value returned by getValue [0...maxIteration]
178  */
179 Ingemi.prototype.setPixelColor = function(pos, value) {
180     if (this.maxIteration > 255) {
181         value = Math.floor(value/this.maxIteration * 255);
182     }
183     this.image.data[pos] = value;
184     this.image.data[pos+1] = value;
185     this.image.data[pos+2] = value;
186     this.image.data[pos+3] = 255;
187 };
188 
189 /**
190  * Increment block- and image-level counters (to deal with asynchronous pixel calculations)
191  * Callback to renderBlock or putImageData when appropriate
192  */
193 Ingemi.prototype.updateCounters = function () {
194     this.renderedPixels += 1;
195     this.renderedPixelsInBlock += 1;
196     if (this.renderedPixels == this.totalPixels) {
197         this.context.putImageData(this.image, 0, 0);
198         this.drawAxes();
199         if (this.onrender) this.onrender();
200     } else if (this.renderedPixelsInBlock === this.blockSize) {
201         this.blockOffset += this.blockSize;
202         this.renderedPixelsInBlock = 0;
203         this.renderBlock();
204     }
205 };
206 
207 /**
208  * Indicate the current frame being rendered. Not yet implemented.
209  */
210 Ingemi.prototype.drawAxes = function () {
211     var margin = 30;
212     this.context.fillStyle = "rgba(255, 255, 255, 1)";
213     this.context.fillRect(0, margin, this.width, 2);
214     this.context.fillRect(margin, 0, 2, this.height);
215 };
216 
217 /**
218  * Log timing statistics
219  */
220 Ingemi.prototype.logStats = function() {
221     console.log(
222         'Rendered ' + this.width + ' * ' + this.height + ' pixels in ' +
223         ((new Date().getTime() - this.timer)/1000) + ' seconds.'
224     );
225 };

228 /**
229  * A helper class to handle viewport modification in Ingemi
230  * @name IngemiZoom
231  * @class IngemiZoom
232  * @constructor
233  * @param {Ingemi} fractal
234  */
235 function IngemiZoom (fractal) {
236     this.fractal = fractal;
237     this.binds();
238 }
239 
240 /**
241  * Bind all UI events
242  */
243 IngemiZoom.prototype.binds = function () {
244     var _this = this;
245     this.fractal.parentDiv.addEventListener("click", function (e) {
246         _this.handleClick(e.offsetX, e.offsetY);
247     });
248 };
249 
250 /**
251  * Handle click: center on (x, y) and zoom 2x
252  * @param {Integer} x
253  * @param {Integer} y
254  */
255 IngemiZoom.prototype.handleClick = function (x, y) {
256     x = x / this.fractal.upscale;
257     y = y / this.fractal.upscale;
258     var targetLeft = x / this.fractal.width - 0.5;
259     var targetTop = y / this.fractal.height - 0.5;
260     this.fractal.offsetLeft += targetLeft * this.fractal.rangeLeft * 3.5;
261     this.fractal.offsetTop += targetTop * this.fractal.rangeTop * 2;
262     this.fractal.rangeLeft /= 2;
263     this.fractal.rangeTop /= 2;
264     this.fractal.offsetLeft += this.fractal.rangeLeft * 1.75;
265     this.fractal.offsetTop += this.fractal.rangeTop;
266     this.fractal.render();
267 };
