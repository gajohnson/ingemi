1 (function (){
  2 /**
  3  * Ingemi(nate) - An iterative Mandelbrot set generator in JavaScript.
  4  * @name Ingemi
  5  * @class Ingemi
  6  * @constructor
  7  * @param {Node} parentDiv The container where Ingemi will render.
  8  * @param {Object} args An argument object to override most defaults. Valid
  9  *     properties are listed below.
 10  * @property {Integer} upscale Used to strech internal canvas dimensions for
 11  *     fast, low-resolution renders.
 12  * @property {Float} rangeLeft Width of coordinate system mapped to the canvas
 13  *     (smaller numbers produce zoom).
 14  * @property {Float} rangeTop Height of coordinate system mapped to the canvas.
 15  * @property {Float} offsetLeft Horizontal offset of the coordinate system
 16  *     relative to the center of the canvas. Units are the same as rangeLeft.
 17  * @property {Float} offsetTop Vertical offset of the coordinate system
 18  *     relative to the center of the canvas.
 19  * @property {Integer} maxIteration Maximum iteration used in escape-velocity
 20  *     calculations. High numbers produce greater color differentiation.
 21  * @property {Integer} blockSize Number of pixels to render concurrently.
 22  *     Higher values may increase performance at the cost of browser stability.
 23  * @property {Function} onrender Callback function fired every time a render
 24  *     is completed.
 25  * @throws Error if parentDiv is not a valid <div>.
 26  */
 27 Ingemi = function (parentDiv, args) {
 28     if (!parentDiv || parentDiv.nodeName != 'DIV') {
 29         throw new Error('You must specify the div where Ingemi will render');
 30     }
 31     args = args || {};
 32     this.parentDiv = parentDiv;
 33     this.upscale = args['upscale'] || 1;
 34     this.rangeLeft = args['rangeLeft'] || 1;
 35     this.rangeTop = args['rangeTop'] || 1;
 36     this.offsetLeft = args['offsetLeft'] || 0;
 37     this.offsetTop = args['offsetTop'] || 0;
 38     this.maxIteration = args['maxIteration'] || 255;
 39     this.blockSize = args['blockSize'] || 2500;
 40     this.onrender = (typeof args['onrender'] === 'function') ? args['onrender'] : null;
 41     this.lock = false;
 42 };
 43 
 44 /**
 45  * Initialize the canvas element.
 46  */
 47 Ingemi.prototype.init = function () {
 48     this.makeCanvas();
 49     this.scaleCanvas();
 50 };
 51 
 52 /**
 53  * Create a canvas in the parent div and inherit its size.
 54  */
 55 Ingemi.prototype.makeCanvas = function () {
 56     this.canvas = document.createElement('canvas');
 57     this.parentDiv.appendChild(this.canvas);
 58     this.context = this.canvas.getContext('2d');
 59     this.clientWidth = this.parentDiv.clientWidth;
 60     this.clientHeight = this.parentDiv.clientHeight;
 61     this.canvas.style.width = this.clientWidth + 'px';
 62     this.canvas.style.height =  this.clientHeight + 'px';
 63 };
 64 
 65 /**
 66  * Set the internal size of the canvas element.
 67  */
 68 Ingemi.prototype.scaleCanvas = function () {
 69     this.width = Math.floor(this.clientWidth / this.upscale);
 70     this.height = Math.floor(this.clientHeight / this.upscale);
 71     this.canvas.width = this.width;
 72     this.canvas.height = this.height;
 73     this.image = this.context.createImageData(this.width, this.height);
 74     this.totalPixels = this.width * this.height;
 75 };
 76 
 77 /**
 78  * Render the entire scene using the current viewport and context.
 79  */
 80 Ingemi.prototype.render = function () {
 81     if (this.lock) return;
 82     this.lock = true;
 83     this.renderedPixels = 0;
 84     this.renderedPixelsInBlock = 0;
 85     this.blockOffset = 0;
 86     this.renderBlock();
 87 };
 88 
 89 /**
 90  * Render one row asynchronously. Increasing blockSize can potentially
 91  *     decrease rendering times at the expense of CPU usage.
 92  */
 93 Ingemi.prototype.renderBlock = function () {
 94     /**
 95      * Prevent pixel calculation beyond the total number of pixel
 96      * @type {Boolean}
 97      */
 98     var overSized = this.blockSize + this.blockOffset > this.totalPixels;
 99     var lim = overSized ? this.totalPixels - this.blockOffset : this.blockSize;
100     var i;
101     for (i = 0; i < lim; i += 1) {
102         var pos = this.blockOffset + i;
103         this.setPixel(pos % this.width, Math.floor(pos/this.width));
104     }
105 };
106 
107 /**
108  * Asynchronously determine and set the value for the specified pixel
109  *     and internally check if we are done rendering either a block or image.
110  * @param {Integer} left
111  * @param {Integer} top
112  */
113 Ingemi.prototype.setPixel = function (left, top) {
114     var _this = this;
115     setTimeout(function () {
116         var pos = _this.gridToLine(left, top) * 4;
117         var value = _this.getValue(left, top);
118         _this.setPixelColor(pos, value);
119         _this.updateCounters();
120     }, 0);
121 };
122 
123 /**
124  * Convert cartesian coordinates to a one-dimensional index.
125  * @param {Integer} left
126  * @param {Integer} top
127  */
128 Ingemi.prototype.gridToLine = function (left, top) {
129     return top * this.width + left;
130 };
131 
132 /**
133  * Main logic for Mandelbrot generation
134  *     - TODO Abstract out for more fractal types.
135  * @param {Integer} left
136  * @param {Integer} top
137  */
138 Ingemi.prototype.getValue = function (left, top) {
139     var scaledX, scaledY;
140     scaledX = this.rangeLeft * 3.5 * left / this.width - 2.5 + this.offsetLeft;
141     scaledY = this.rangeTop * 2 * top / this.height - 1 + this.offsetTop;
142     /** Optimize against inner cartoid and return known maxIteration */
143     if (this.isInCartoid(scaledX, scaledY)) {
144         return this.maxIteration;
145     }
146     var x = 0;
147     var y = 0;
148     var iteration = 0;
149     var xtemp;
150     while (x*x + y*y < 4 && iteration < this.maxIteration) {
151         xtemp = x*x - y*y + scaledX;
152         y = 2*x*y + scaledY;
153         x = xtemp;
154         iteration += 1;
155     }
156     return iteration;
157 };
158 
159 /**
160  * Simple optimization to prevent computing to maximum iteration in the center
161  *     of the unzoomed Mandelbrot set.
162  * @param {Float} left
163  * @param {Float} top
164  */
165 Ingemi.prototype.isInCartoid = function (left, top) {
166     var p = Math.pow(left - 0.25, 2) + (top * top);
167     var q = Math.sqrt(p);
168     return left <= q - (2 * p) + 0.25;
169 };
170 
171 /**
172  * Map an integer [0...maxIteration] into some rgb spectrum
173  *     and write it to the imageData array.
174  * @param {Integer} pos The linear position of the pixel being modified
175  * @param {Integer} value The value returned by getValue [0...maxIteration]
176  */
177 Ingemi.prototype.setPixelColor = function(pos, value) {
178     if (this.maxIteration > 255) {
179         value = Math.floor(value/this.maxIteration * 255);
180     }
181     this.image.data[pos] = value;
182     this.image.data[pos+1] = value;
183     this.image.data[pos+2] = value;
184     this.image.data[pos+3] = 255;
185 };
186 
187 /**
188  * Increment block- and image-level counters (to deal with asynchronous pixel
189  *     calculations).
190  */
191 Ingemi.prototype.updateCounters = function () {
192     this.renderedPixels += 1;
193     this.renderedPixelsInBlock += 1;
194     if (this.renderedPixels == this.totalPixels) {
195         this.finalize();
196     } else if (this.renderedPixelsInBlock === this.blockSize) {
197         this.nextBlock();
198     }
199 };
200 
201 /**
202  * Increment the global block offset, reset the block-level counter, and make
203  *     rendering call.
204  */
205 Ingemi.prototype.nextBlock = function () {
206     this.blockOffset += this.blockSize;
207     this.renderedPixelsInBlock = 0;
208     this.renderBlock();
209 };
210 
211 /**
212  * Draw current image to canvas, unlock future rendering calls, and finally
213  *     call onrender
214  */
215 Ingemi.prototype.finalize = function () {
216     this.context.putImageData(this.image, 0, 0);
217     this.lock = false;
218     if (this.onrender) this.onrender();
219 };
220 
221 /**
222  * Center the viewport on (x, y) and zoom in 2x.
223  * @param {Integer} x In pixels from the left of the canvas
224  * @param {Integer} y In pixels from the top of the canvas
225  */
226 Ingemi.prototype.zoom = function (x, y) {
227     x = x / this.upscale;
228     y = y / this.upscale;
229     var targetLeft = x / this.width - 0.5;
230     var targetTop = y / this.height - 0.5;
231     this.offsetLeft += targetLeft * this.rangeLeft * 3.5;
232     this.offsetTop += targetTop * this.rangeTop * 2;
233     this.rangeLeft /= 2;
234     this.rangeTop /= 2;
235     this.offsetLeft += this.rangeLeft * 1.75;
236     this.offsetTop += this.rangeTop;
237     this.render();
238 };

241  * @export Ingemi as window.Ingemi
242  */
243 window['Ingemi'] = Ingemi;

247  */
248 Ingemi.prototype['init'] = Ingemi.prototype.init;
249 
250 /**

252  */
253 Ingemi.prototype['render'] = Ingemi.prototype.render;
254 
255 /**
256  * @export Ingemi.prototype.zoom as window.Ingemi.prototype.zoom
257  */
258 Ingemi.prototype['zoom'] = Ingemi.prototype.zoom;
259 })();
